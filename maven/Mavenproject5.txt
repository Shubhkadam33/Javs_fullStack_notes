public List<Student>getallstudents(){

    		s.setFullname(rs.getString("fullname"));
    		s.setEmail(rs.getString("email"));
    	
    		s.setPassword(rs.getString("password"));
    		s.setAddress(rs.getString("address"));
    		return s;
    	});
    }
