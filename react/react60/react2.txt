>we can't you for attribute becuse for is a reserve keyword in js

so we use alternative word that is "htmlfor"

>if we try to write for then we will get error
export default function Form(){ 
  let [fullName,setFullName]=useState("shubham");
  
  let handleNameChange=(event)=>{
   setFullName(event.target.value);
};

return(
  <form>
     <label htmlFor="username">Full Name</label>
     <input placeholder="enter full name" type="text" value={fullName} onChange={handleNameChange} id="username" />
     <button>submit</button>
   </form>
)
